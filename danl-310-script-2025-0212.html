# n(): to count the number of observations for each group

bigregion_religion <- gss_sm |> 
  select(id, bigregion, religion) |> 
  arrange(bigregion, religion, id) |> 
  group_by(bigregion, religion) |> 
  summarize(n = n()) |> 
  mutate(tot = sum(n) ) |> 
  mutate(pct = 100 * n / tot)
